f_trim_compact(string s) =>
    str.replace_all(str.replace_all(s, " ", ""), "\t", "")

    string t = f_trim_compact(s)

    na(n) ? na : int(math.round(n))

f_register_merge_range(int a, int b, int n, array<int> mergeStart, array<int> mergeEnd) =>
    int aa = math.max(1, math.min(a, n))
    int bb = math.max(1, math.min(b, n))
    if aa > bb
        int tmp = aa
        aa := bb
        bb := tmp

    bool overlap = false
    for k = aa to bb
        if array.get(mergeStart, k) != 0 or array.get(mergeEnd, k) != 0
            overlap := true

    if not overlap
        array.set(mergeStart, aa, aa)
        array.set(mergeEnd, aa, bb)
        if bb > aa
            for k = aa + 1 to bb
                array.set(mergeStart, k, -1)
                array.set(mergeEnd, k, -1)

f_apply_default_singles(int n, array<int> mergeStart, array<int> mergeEnd) =>
    for i = 1 to n
        if array.get(mergeStart, i) == 0 and array.get(mergeEnd, i) == 0
            array.set(mergeStart, i, i)
            array.set(mergeEnd, i, i)

f_parse_merge_groups(string txt, int n, array<int> mergeStart, array<int> mergeEnd) =>
    array.fill(mergeStart, 0)
    array.fill(mergeEnd, 0)

    string clean = f_trim_compact(txt)
    clean := str.replace_all(clean, ",", ";")

    if str.length(clean) == 0
        f_apply_default_singles(n, mergeStart, mergeEnd)
    else
        array<string> groups = str.split(clean, ";")
        int gc = array.size(groups)
        if gc > 0
            for gi = 0 to gc - 1
                string g = array.get(groups, gi)
                if str.length(g) == 0
                    continue
                int dashPos = str.pos(g, "-")
                if na(dashPos)
                    int x = f_parse_int(g)
                    if not na(x)
                        f_register_merge_range(x, x, n, mergeStart, mergeEnd)
                else
                    string aStr = str.substring(g, 0, dashPos)
                    string bStr = str.substring(g, dashPos + 1)
                    int a = f_parse_int(aStr)
                    int b = f_parse_int(bStr)
                    if not na(a) and not na(b)
                        f_register_merge_range(a, b, n, mergeStart, mergeEnd)

        f_apply_default_singles(n, mergeStart, mergeEnd)

f_build_profile(int blockStart, int blockEnd, array<int> srcTime, array<float> srcOpen, array<float> srcLow, array<float> srcHigh, array<float> srcClose, array<float> srcVolume) =>
    float lo = 1e20
    float hi = -1e20
    bool found = false

    int srcCount = array.size(srcTime)
    if srcCount > 0
        for i = 0 to srcCount - 1
            int t = array.get(srcTime, i)
            if t < blockStart
                continue
            if t >= blockEnd
                break
            lo := math.min(lo, array.get(srcLow, i))
            hi := math.max(hi, array.get(srcHigh, i))
            found := true

    bool ok = found and hi > lo
    float binSize = na
    int bins = 0
    array<float> hist = array.new_float()
    array<float> histUp = array.new_float()
    array<float> histDown = array.new_float()
    float totalVol = 0.0
    float totalUp = 0.0
    float totalDown = 0.0
    int pocIdx = 0
    int vaLow = 0
    int vaHigh = 0
    float pocPrice = na
    float vahPrice = na
    float valPrice = na

    if ok
        float minStep = math.max(syminfo.mintick, inpMinBinSizeTicks * syminfo.mintick)
        float adaptive = (hi - lo) / math.max(10, inpMaxBinsPerDay)
        binSize := math.max(adaptive, minStep)
        bins := math.max(1, math.min(int(math.ceil((hi - lo) / binSize)), math.max(10, inpMaxBinsPerDay)))

        hist := array.new_float(bins, 0.0)
        histUp := array.new_float(bins, 0.0)
        histDown := array.new_float(bins, 0.0)

        if srcCount > 0
            for i = 0 to srcCount - 1
                int t = array.get(srcTime, i)
                if t < blockStart
                    continue
                if t >= blockEnd
                    break

                float v = array.get(srcVolume, i)
                if v <= 0
                    continue

                float o = array.get(srcOpen, i)
                float h = array.get(srcHigh, i)
                float l = array.get(srcLow, i)
                float c = array.get(srcClose, i)
                float upRatio = c > o ? 1.0 : c < o ? 0.0 : 0.5
                float upV = v * upRatio
                float downV = v - upV

                // Keep totals directly from 5m source bars so delta does not depend on histogram binning.
                totalVol += v
                totalUp += upV
                totalDown += downV

                if not inpUseBarRangeAllocation or h <= l
                    float px = (h + l + c) / 3.0
                    int idx = int(math.floor((px - lo) / binSize))
                    idx := math.max(0, math.min(idx, bins - 1))
                    array.set(hist, idx, array.get(hist, idx) + v)
                    array.set(histUp, idx, array.get(histUp, idx) + upV)
                    array.set(histDown, idx, array.get(histDown, idx) + downV)
                else
                    int loBin = int(math.floor((l - lo) / binSize))
                    int hiBin = int(math.floor((h - lo) / binSize))
                    loBin := math.max(0, loBin)
                    hiBin := math.min(bins - 1, hiBin)
                    hiBin := math.max(hiBin, loBin)
                    float pr = h - l
                    for k = loBin to hiBin
                        float bLo = lo + k * binSize
                        float bHi = bLo + binSize
                        float overlap = math.max(0.0, math.min(h, bHi) - math.max(l, bLo))
                        if overlap <= 0
                            continue
                        float w = overlap / pr
                        array.set(hist, k, array.get(hist, k) + v * w)
                        array.set(histUp, k, array.get(histUp, k) + upV * w)
                        array.set(histDown, k, array.get(histDown, k) + downV * w)

        float maxVol = -1.0
        float histTotal = 0.0
        for i = 0 to bins - 1
            float hv = array.get(hist, i)
            histTotal += hv
            if hv > maxVol
                maxVol := hv
                pocIdx := i

        ok := totalVol > 0 and histTotal > 0

        if ok
            float target = histTotal * (inpValueAreaPercent / 100.0)
            vaLow := pocIdx
            vaHigh := pocIdx
            float acc = array.get(hist, pocIdx)

            while acc < target
                float dn = na
                float up = na
                if vaLow - 1 >= 0
                    dn := array.get(hist, vaLow - 1)
                if vaHigh + 1 < bins
                    up := array.get(hist, vaHigh + 1)

                if na(dn) and na(up)
                    break

                if na(dn)
                    vaHigh += 1
                    acc += array.get(hist, vaHigh)
                else if na(up)
                    vaLow -= 1
                    acc += array.get(hist, vaLow)
                else if up > dn
                    vaHigh += 1
                    acc += array.get(hist, vaHigh)
                else
                    vaLow -= 1
                    acc += array.get(hist, vaLow)

            pocPrice := lo + (pocIdx + 0.5) * binSize
            vahPrice := lo + (vaHigh + 0.5) * binSize
            valPrice := lo + (vaLow + 0.5) * binSize
            pocPrice := f_price_snap(pocPrice)
            vahPrice := f_price_snap(vahPrice)
            valPrice := f_price_snap(valPrice)

    [ok, lo, hi, binSize, bins, totalVol, totalUp, totalDown, pocIdx, vaLow, vaHigh, pocPrice, vahPrice, valPrice, hist, histUp, histDown]

f_draw_profile(int dayStart, int dayEnd, int levelsEndTime, float dayLow, float binSize, int bins, int vaLow, int vaHigh, float pocPrice, float vahPrice, float valPrice, array<float> hist, array<float> histUp, array<float> histDown, float totalUp, float totalDown, string labelSuffix, bool drawMainLabels) =>
    bool drawHistogram = inpRenderMode == "Histogram" or inpRenderMode == "Both"
    bool drawLine = inpRenderMode == "Line" or inpRenderMode == "Both"

    float maxVol = 0.0
    float minNonZeroVol = 1e20
    for i = 0 to bins - 1
        float hv = array.get(hist, i)
        maxVol := math.max(maxVol, hv)
        if hv > 0
            minNonZeroVol := math.min(minNonZeroVol, hv)

    if maxVol > 0
        int tStart = dayStart
        int tEnd = dayEnd
        int levelsEnd = levelsEndTime
        if tEnd <= tStart
            tEnd := tStart + BASE_PROFILE_TF_SEC * 1000
        if levelsEnd <= tStart
            levelsEnd := tEnd

        int blockMs = math.max(BASE_PROFILE_TF_SEC * 1000, tEnd - tStart)
        float widthFrac = math.max(0.02, math.min(0.95, inpProfileWidthPercent / 100.0))
        int widthMs = math.max(BASE_PROFILE_TF_SEC * 1000, int(math.round(blockMs * widthFrac)))

        int leftLimit = na
        int rightLimit = na

        if not inpDrawRightToLeft
            leftLimit := tStart
            rightLimit := math.min(tEnd, tStart + widthMs)
        else
            leftLimit := math.max(tStart, tEnd - widthMs)
            rightLimit := tEnd

        float minFrac = math.max(0.0, math.min(0.20, inpMinBinWidthPercent / 100.0))
        int profileSpan = math.max(BASE_PROFILE_TF_SEC * 1000, rightLimit - leftLimit)

        bool hasPrevTip = false
        int prevTipT = na
        float prevTipP = na

        for i = 0 to bins - 1
            float h = array.get(hist, i)
            if h <= 0
                continue

            float frac = 1.0
            if maxVol > minNonZeroVol
                frac := (h - minNonZeroVol) / (maxVol - minNonZeroVol)
            frac := math.max(0.0, math.min(1.0, frac))
            if minFrac > 0
                frac := minFrac + frac * (1.0 - minFrac)

            int t1 = na
            int t2 = na

            if not inpDrawRightToLeft
                t1 := leftLimit
                t2 := leftLimit + int(math.round(profileSpan * frac))
                t2 := math.min(rightLimit, t2)
                if t2 <= t1
                    t2 := math.min(rightLimit, t1 + 1)
            else
                t1 := rightLimit - int(math.round(profileSpan * frac))
                t1 := math.max(leftLimit, t1)
                t2 := rightLimit
                if t2 <= t1
                    t1 := math.max(leftLimit, t2 - 1)

            float p1 = dayLow + i * binSize
            float p2 = dayLow + (i + 1) * binSize
            p1 := f_price_snap(p1)
            p2 := f_price_snap(p2)
            color binColor = i >= vaLow and i <= vaHigh ? inpProfileColorInsideVA : inpProfileColorOutsideVA

            if drawHistogram
                box bx = box.new(left=t1, top=p2, right=t2, bottom=p1, xloc=xloc.bar_time, border_color=color.new(binColor, inpHistogramBorderTransparency), bgcolor=color.new(binColor, inpHistogramFillTransparency))
                array.push(gBoxes, bx)

            int tipT = not inpDrawRightToLeft ? t2 : t1
            float tipP = f_price_snap(dayLow + (i + 0.5) * binSize)

            if drawLine
                if hasPrevTip
                    line ln = line.new(x1=prevTipT, y1=prevTipP, x2=tipT, y2=tipP, xloc=xloc.bar_time, color=inpContourColor, style=lineStyleContour, width=inpContourWidth)
                    array.push(gLines, ln)
                hasPrevTip := true
                prevTipT := tipT
                prevTipP := tipP

            if inpShowBidAskText
                float upBin = array.get(histUp, i)
                float downBin = array.get(histDown, i)
                int textT = not inpDrawRightToLeft ? tipT + BASE_PROFILE_TF_SEC * 1000 : tipT - BASE_PROFILE_TF_SEC * 1000
                f_make_text(textT, tipP, f_volume_mode_text(downBin, upBin, false), color.white)

        float pocY = f_price_snap(pocPrice)
        float vahY = f_price_snap(vahPrice)
        float valY = f_price_snap(valPrice)
        line poc = line.new(x1=tStart, y1=pocY, x2=levelsEnd, y2=pocY, xloc=xloc.bar_time, color=inpPOCColor, style=lineStylePOC, width=inpPOCWidth)
        line vah = line.new(x1=tStart, y1=vahY, x2=levelsEnd, y2=vahY, xloc=xloc.bar_time, color=inpVAHColor, style=lineStyleVA, width=inpVAWidth)
        line val = line.new(x1=tStart, y1=valY, x2=levelsEnd, y2=valY, xloc=xloc.bar_time, color=inpVALColor, style=lineStyleVA, width=inpVAWidth)
        array.push(gLines, poc)
        array.push(gLines, vah)
        array.push(gLines, val)

        if inpShowLabels and drawMainLabels
            int txtTime = levelsEnd
            string sfx = str.length(labelSuffix) > 0 ? " " + labelSuffix : ""
            f_make_label(txtTime, pocY, inpPOCLabelText == "" ? "POC" + sfx : inpPOCLabelText + sfx, inpPOCColor)
            f_make_label(txtTime, vahY, inpVAHLabelText == "" ? "VAH" + sfx : inpVAHLabelText + sfx, inpVAHColor)
            f_make_label(txtTime, valY, "VAL" + sfx, inpVALColor)

        if inpShowBidAskTotals
            float profileHigh = dayLow + bins * binSize
            float yPad = math.max((profileHigh - dayLow) * 0.04, syminfo.mintick * 10.0)
            float totalY = f_price_snap(dayLow - yPad)
            int totalX = not inpDrawRightToLeft ? rightLimit + BASE_PROFILE_TF_SEC * 1000 : leftLimit - BASE_PROFILE_TF_SEC * 1000
            f_make_text(totalX, totalY, f_volume_mode_text(totalDown, totalUp, true), color.white)

f_draw_previous_profile_on_today(int todayStart, int todayEnd, float pocPrice, float vahPrice, float valPrice) =>
    int tEnd = todayEnd
    if tEnd <= todayStart
        tEnd := todayStart + DAY_MS

    float pocY = f_price_snap(pocPrice)
    float vahY = f_price_snap(vahPrice)
    float valY = f_price_snap(valPrice)
    line poc = line.new(x1=todayStart, y1=pocY, x2=tEnd, y2=pocY, xloc=xloc.bar_time, color=inpPOCColor, style=lineStylePOC, width=inpPOCWidth)
    line vah = line.new(x1=todayStart, y1=vahY, x2=tEnd, y2=vahY, xloc=xloc.bar_time, color=inpVAHColor, style=lineStyleVA, width=inpVAWidth)
    line val = line.new(x1=todayStart, y1=valY, x2=tEnd, y2=valY, xloc=xloc.bar_time, color=inpVALColor, style=lineStyleVA, width=inpVAWidth)
    array.push(gLines, poc)
    array.push(gLines, vah)
    array.push(gLines, val)

    if inpShowLabels
        int txtTime = tEnd
        f_make_label(txtTime, pocY, "Prev " + (inpPOCLabelText == "" ? "POC" : inpPOCLabelText), inpPOCColor)
        f_make_label(txtTime, vahY, "Prev " + (inpVAHLabelText == "" ? "VAH" : inpVAHLabelText), inpVAHColor)
        f_make_label(txtTime, valY, "Prev VAL", inpVALColor)

if barstate.islast
    string buildSig = syminfo.tickerid + "|" + timeframe.period + "|" + sessionTz + "|" + dayTz + "|" + str.tostring(inpDaysToDraw) + "|" + str.tostring(inpIncludeCurrentDay) + "|" + str.tostring(inpValueAreaPercent) + "|" + str.tostring(inpMaxBinsPerDay) + "|" + str.tostring(inpMinBinSizeTicks) + "|" + str.tostring(inpProfileWidthPercent) + "|" + str.tostring(inpUseBarRangeAllocation) + "|" + str.tostring(inpIgnoreSundaySession) + "|" + str.tostring(inpShowLabels) + "|" + inpPOCLabelText + "|" + inpVAHLabelText + "|" + inpVolumeTextMode + "|" + str.tostring(inpShowBidAskText) + "|" + str.tostring(inpShowBidAskTotals) + "|" + str.tostring(inpUpDownTextSize) + "|" + str.tostring(inpHistogramFillTransparency) + "|" + str.tostring(inpHistogramBorderTransparency) + "|" + str.tostring(inpMinBinWidthPercent) + "|" + inpRenderMode + "|" + str.tostring(inpDrawRightToLeft) + "|" + str.tostring(inpExtendMergedLevelsToToday) + "|" + inpMergeGroups + "|" + f_color_sig(inpProfileColorOutsideVA) + "|" + f_color_sig(inpContourColor) + "|" + f_color_sig(inpProfileColorInsideVA) + "|" + f_color_sig(inpPOCColor) + "|" + f_color_sig(inpVAHColor) + "|" + f_color_sig(inpVALColor) + "|" + str.tostring(inpPOCWidth) + "|" + str.tostring(inpVAWidth) + "|" + inpPOCStyleOpt + "|" + inpVAStyleOpt + "|" + str.tostring(inpContourWidth) + "|" + inpContourStyleOpt + "|" + str.tostring(inpMaxScanBars) + "|" + str.tostring(inpUseSessionFilter) + "|" + inpSessionHours + "|" + inpSessionTimezoneOpt + "|" + inpDayTimezoneOpt
    int currentBaseTime = na(srcTimeSeries) ? int(time) : int(srcTimeSeries)
    bool configChanged = buildSig != gLastBuildSig
    bool baseChanged = na(gLastBuiltBaseTime) or currentBaseTime != gLastBuiltBaseTime
    bool shouldRebuild = configChanged or baseChanged

    if shouldRebuild
        gLastBuildSig := buildSig
        gLastBuiltBaseTime := currentBaseTime
        f_clear_objects()

        array<int> srcTimes = array.new_int()
        array<int> srcDayStarts = array.new_int()
        array<float> srcOpens = array.new_float()
        array<float> srcLows = array.new_float()
        array<float> srcHighs = array.new_float()
        array<float> srcCloses = array.new_float()
        array<float> srcVolumes = array.new_float()

        int scanBars = math.min(bar_index, math.min(inpMaxScanBars, 5000))

        for k = 0 to scanBars
            int i = scanBars - k
            int chunkSize = 0

            if chartAboveBaseTF
                array<int> tChunk = srcTimeLowerTF[i]
                array<int> dChunk = srcDayLowerTF[i]
                array<float> opChunk = srcOpenLowerTF[i]
                array<float> loChunk = srcLowLowerTF[i]
                array<float> hiChunk = srcHighLowerTF[i]
                array<float> clChunk = srcCloseLowerTF[i]
                array<float> vChunk = srcVolumeLowerTF[i]
                array<float> sessChunk = srcSessionLowerTF[i]
                bool haveChunks = not na(tChunk) and not na(dChunk) and not na(opChunk) and not na(loChunk) and not na(hiChunk) and not na(clChunk) and not na(vChunk) and not na(sessChunk)

                if haveChunks
                    chunkSize := array.size(tChunk)

                if haveChunks and chunkSize > 0
                    for j = 0 to chunkSize - 1
                        float inSession = array.get(sessChunk, j)
                        if inpUseSessionFilter and (na(inSession) or inSession < 0.5)
                            continue

                        int t = int(array.get(tChunk, j))
                        int ds = int(array.get(dChunk, j))
                        float op = array.get(opChunk, j)
                        float lo = array.get(loChunk, j)
                        float hi = array.get(hiChunk, j)
                        float cl = array.get(clChunk, j)
                        float v = array.get(vChunk, j)
                        if na(t) or na(ds) or na(op) or na(lo) or na(hi) or na(cl) or na(v)
                            continue
                        f_push_source_bar(srcTimes, srcDayStarts, srcOpens, srcLows, srcHighs, srcCloses, srcVolumes, t, ds, op, lo, hi, cl, v)

            if chunkSize == 0 and not chartAboveBaseTF
                float inSession = srcSessionSeries[i]
                if inpUseSessionFilter and (na(inSession) or inSession < 0.5)
                    continue

                int t = int(srcTimeSeries[i])
                int ds = int(srcDaySeries[i])
                float op = srcOpenSeries[i]
                float lo = srcLowSeries[i]
                float hi = srcHighSeries[i]
                float cl = srcCloseSeries[i]
                float v = srcVolumeSeries[i]
                if na(t) or na(ds) or na(op) or na(lo) or na(hi) or na(cl) or na(v)
                    continue
                f_push_source_bar(srcTimes, srcDayStarts, srcOpens, srcLows, srcHighs, srcCloses, srcVolumes, t, ds, op, lo, hi, cl, v)

        int srcRawCount = array.size(srcTimes)
        int keepCount = math.min(inpMaxScanBars, 5000)
        if srcRawCount > keepCount
            int keepFrom = srcRawCount - keepCount
            array<int> trimmedTimes = array.new_int()
            array<int> trimmedDayStarts = array.new_int()
            array<float> trimmedOpens = array.new_float()
            array<float> trimmedLows = array.new_float()
            array<float> trimmedHighs = array.new_float()
            array<float> trimmedCloses = array.new_float()
            array<float> trimmedVolumes = array.new_float()

            for r = keepFrom to srcRawCount - 1
                array.push(trimmedTimes, array.get(srcTimes, r))
                array.push(trimmedDayStarts, array.get(srcDayStarts, r))
                array.push(trimmedOpens, array.get(srcOpens, r))
                array.push(trimmedLows, array.get(srcLows, r))
                array.push(trimmedHighs, array.get(srcHighs, r))
                array.push(trimmedCloses, array.get(srcCloses, r))
                array.push(trimmedVolumes, array.get(srcVolumes, r))

            srcTimes := trimmedTimes
            srcDayStarts := trimmedDayStarts
            srcOpens := trimmedOpens
            srcLows := trimmedLows
            srcHighs := trimmedHighs
            srcCloses := trimmedCloses
            srcVolumes := trimmedVolumes

        int srcCount = array.size(srcTimes)

        if srcCount == 0
            f_make_label(int(time), close, "No 5-minute data available", color.orange)
        else
            array<int> uniqStarts = array.new_int()
            int todayStart = array.get(srcDayStarts, srcCount - 1)
            int latestSrcTime = array.get(srcTimes, srcCount - 1)

            for k = 0 to srcCount - 1
                int i = srcCount - 1 - k
                int ds = array.get(srcDayStarts, i)
                if array.size(uniqStarts) == 0 or ds != array.get(uniqStarts, array.size(uniqStarts) - 1)
                    array.push(uniqStarts, ds)
                if array.size(uniqStarts) >= inpDaysToDraw + 30
                    break

            array<int> baseStarts = array.new_int()
            array<int> baseEnds = array.new_int()

            int usz = array.size(uniqStarts)
            if usz > 0
                for idx = 0 to usz - 1
                    int ds = array.get(uniqStarts, idx)
                    if not inpIncludeCurrentDay and ds == todayStart
                        continue
                    if inpIgnoreSundaySession and dayofweek(ds) == dayofweek.sunday
                        continue

                    int de = ds + DAY_MS

                    if de > ds
                        array.push(baseStarts, ds)
                        array.push(baseEnds, de)

                    if array.size(baseStarts) >= inpDaysToDraw
                        break

            int n = array.size(baseStarts)

            if n > 0
                array<int> mergeStart = array.new_int(n + 1, 0)
                array<int> mergeEnd = array.new_int(n + 1, 0)
                f_parse_merge_groups(inpMergeGroups, n, mergeStart, mergeEnd)

                bool capturedTodayGroup = false
                bool havePrevOfToday = false
                float prevPoc = na
                float prevVah = na
                float prevVal = na

                for i = 1 to n
                    int ms = array.get(mergeStart, i)
                    int me = array.get(mergeEnd, i)
                    if ms == -1 and me == -1
                        continue
                    if ms <= 0 or me <= 0
                        continue

                    int a = ms
                    int b = me
                    int blockStart = array.get(baseStarts, b - 1)
                    int blockEnd = array.get(baseEnds, a - 1)

                    [ok, lo, hi, binSize, bins, totalVol, totalUp, totalDown, pocIdx, vaLow, vaHigh, pocPrice, vahPrice, valPrice, hist, histUp, histDown] = f_build_profile(blockStart, blockEnd, srcTimes, srcOpens, srcLows, srcHighs, srcCloses, srcVolumes)

                    if ok
                        bool isMergedBlock = b > a
                        int levelsEnd = blockEnd
                        if inpExtendMergedLevelsToToday and isMergedBlock
                            levelsEnd := math.max(levelsEnd, latestSrcTime)
                        bool isPrevOfTodayGroup = inpIncludeCurrentDay and capturedTodayGroup and not havePrevOfToday
                        string lbl = f_range_label(a, b)
                        f_draw_profile(blockStart, blockEnd, levelsEnd, lo, binSize, bins, vaLow, vaHigh, pocPrice, vahPrice, valPrice, hist, histUp, histDown, totalUp, totalDown, lbl, not isPrevOfTodayGroup)

                        if inpIncludeCurrentDay
                            if not capturedTodayGroup and a == 1
                                capturedTodayGroup := true
                            else if capturedTodayGroup and not havePrevOfToday
                                prevPoc := pocPrice
                                prevVah := vahPrice
                                prevVal := valPrice
                                havePrevOfToday := true

                if inpIncludeCurrentDay and havePrevOfToday
                    f_draw_previous_profile_on_today(todayStart, todayStart + DAY_MS, prevPoc, prevVah, prevVal)
